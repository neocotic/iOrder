                 ¶

      url = chrome.extension.getURL('pages/options.html');

      chrome.tabs.create({url: ORDERS_URL});
      break;
    case 'view':
      order = ext.getOrder(request.data.number, request.data.code);
      if (order) chrome.tabs.create({url: getOrderUrl(order)});
      break;
